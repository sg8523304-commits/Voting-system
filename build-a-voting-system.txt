// Initialize the poll as a Map
const poll = new Map();

// Function to add an option
function addOption(option) {
  if (!option || option.trim() === "") {
    return "Option cannot be empty.";
  }
  if (poll.has(option)) {
    return `Option "${option}" already exists.`;
  }
  poll.set(option, new Set());
  return `Option "${option}" added to the poll.`;
}

// Function to vote
function vote(option, voterId) {
  if (!poll.has(option)) {
    return `Option "${option}" does not exist.`;
  }
  const voters = poll.get(option);
  if (voters.has(voterId)) {
    return `Voter ${voterId} has already voted for "${option}".`;
  }
  voters.add(voterId);
  return `Voter ${voterId} voted for "${option}".`;
}

// Function to display poll results
function displayResults() {
  let result = "Poll Results:\n";
  for (const [option, voters] of poll.entries()) {
    result += `${option}: ${voters.size} votes\n`;
  }
  return result.trim(); // remove last newline
}

/* ----- Example usage ----- */

// Add options
console.log(addOption("Turkey"));   // Option "Turkey" added to the poll.
console.log(addOption("Morocco"));  // Option "Morocco" added to the poll.
console.log(addOption("Italy"));    // Option "Italy" added to the poll.

// Add duplicate / empty
console.log(addOption("Turkey"));   // Option "Turkey" already exists.
console.log(addOption(""));         // Option cannot be empty.

// Voting
console.log(vote("Turkey", 1));     // Voter 1 voted for "Turkey".
console.log(vote("Morocco", 2));    // Voter 2 voted for "Morocco".
console.log(vote("Turkey", 3));     // Voter 3 voted for "Turkey".
console.log(vote("Turkey", 1));     // Voter 1 has already voted for "Turkey".
console.log(vote("Italy", 4));      // Voter 4 voted for "Italy".
console.log(vote("Spain", 5));      // Option "Spain" does not exist.

// Display results
console.log(displayResults());
